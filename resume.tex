  \vspace{1pt}%
  \noindent\textbf{#1}\hfill{#2}\\[1pt]%
  \noindent{#3}\hfill{#4}%
  \vspace{1pt}%
}

% \eduentry{左列学校}{中列专业学位}{右列时间}{右列地点（可选）}
\newcommand{\eduentry}[4]{%
  \vspace{1pt}%
  \noindent
  \begin{minipage}[t]{0.20\linewidth}
    \raggedright\textbf{#1}
  \end{minipage}%
  \begin{minipage}[t]{0.60\linewidth}
    \centering\textbf{#2}
  \end{minipage}%
  \begin{minipage}[t]{0.20\linewidth}
    \raggedleft #3\if\relax\detokenize{#4}\relax\else\quad #4\fi
  \end{minipage}%
  \vspace{0pt}%
}

% \workentry{左列公司}{中列岗位}{右列时间}{右列地点（可选）}

  \end{minipage}%
  \begin{minipage}[t]{0.40\linewidth}
    \raggedleft #3\if\relax\detokenize{#4}\relax\else\quad #4\fi
  \end{minipage}%
  \vspace{0pt}%
}
